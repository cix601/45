const stops = [ { name: "Stop A", url: "https://www3.septa.org/hackathon/BusSchedules/?req1=16601" }, { name: "Stop B", url: "https://www3.septa.org/hackathon/BusSchedules/?req1=16131" }, { name: "Stop C", url: "https://www3.septa.org/hackathon/BusSchedules/?req1=16129" } ]; const busData = {}; function fetchBusData() { const promises = stops.map(stop => fetch(stop.url) .then(response => response.json()) .then(data => { busData[stop.name] = data; }) .catch(error => console.error(`Error fetching data for ${stop.name}:`, error)) ); return Promise.all(promises); } function calculateTimeDifference(schedule) { const now = new Date(); const scheduledDate = new Date(schedule); const difference = scheduledDate - now; const minutesDifference = Math.ceil(difference / (1000 * 60)); return minutesDifference <= 0 ? "Now" : `${minutesDifference} 分钟`; } function displayWebpage() { const container = document.createElement('div'); document.body.appendChild(container); const stopAData = busData["Stop A"]; if (stopAData) { const routeKeys = Object.keys(stopAData); if (routeKeys.length > 0) { const nextBus = stopAData[routeKeys[0]][0]; const timeLeft = calculateTimeDifference(nextBus.DateCalender); const stopAText = document.createElement('p'); stopAText.textContent = `去唐人街: ${timeLeft} 分钟`; container.appendChild(stopAText); } } const stopBData = busData["Stop B"]; if (stopBData) { const routeKeys = Object.keys(stopBData); if (routeKeys.length > 0) { const nextBus = stopBData[routeKeys[0]][0]; const timeLeft = calculateTimeDifference(nextBus.DateCalender); const stopBText = document.createElement('p'); stopBText.textContent = `从餐馆回家: ${timeLeft} 分钟`; container.appendChild(stopBText); } } const stopCData = busData["Stop C"]; if (stopCData) { const routeKeys = Object.keys(stopCData); if (routeKeys.length > 0) { const nextBus = stopCData[routeKeys[0]][0]; const timeLeft = calculateTimeDifference(nextBus.DateCalender); const stopCText = document.createElement('p'); stopCText.textContent = `从唐人街回家: ${timeLeft} 分钟`; container.appendChild(stopCText); } } } fetchBusData().then(displayWebpage);
